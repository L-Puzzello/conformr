  • country-year panels

## here's a data_map I prepared earlier

toy_AB$data_map
## try convert on the data_map
toy_AB$data_map %>%
  conformr::convert(., std_A, std_B, weight, A_100)

## oops, missing groups
toy_AB$data_map %>%
  dplyr::group_by(country) %>%
  conformr::convert(., std_A, std_B, weight, A_100)

## what if weights are wrong?
toy_AB$data_map %>%
  dplyr::group_by(country) %>%
